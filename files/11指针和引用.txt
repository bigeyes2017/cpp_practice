#include <stdio.h>
int main() {
    int a = 1, c = 2, d[2][3] = { 9,8,7,6 }, e[2] = { 11,12 };
    int& ra = a;
    printf("%10d,%10d,%10d\n", ra, &a, &ra);  //是同一个东西，无需分配内存

    ra = c;
    printf("%10d,%10d,%10d\n", ra, &c, &ra);  //初始化只有在定义时有一次。后面只能使用。

   
    int(*pe)[2];        //数组指针
    pe = &e;
    printf("%10d\n", (*pe)[1]);

    int(*pd)[2][3];     //数组指针
    pd = &d;
    printf("%10d\n", (*pd)[1][1]);

    int* pe2;
    pe2 = e;
    printf("%10d\n", pe2[1]);


    int(*pd2)[3];       //少一维，来指示数组下标.指示的是行优先，每一行有仨数。
    pd2 = d;
    printf("   数  为 :%10d，%10d\n", (pd2)[1][0],pd2[1][1]);
    printf("   地址为 :%10d，%10d\n", pd2, pd2+1);

   
    int* arr[3];        //指针数组，里面全是指针。所以，可以用二维指针parr来指示数组下标。
    int** parr = arr;   //*p = a[0]仍是一个指针。
    printf("   地址为 :%10d，%10d，%10d\n", parr, parr + 1,parr+2);


    return 0;
}
